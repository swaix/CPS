\begin{card}[frametitle={Flusso Diagnostico per Esercizi di Estrazione}]
\textbf{Teoria:} Usa questo schema per decidere l'approccio corretto ai problemi con le urne.
\vspace{10pt}
\begin{itemize}
    \item[\textbf{Q1:}] L'estrazione è \textbf{con reinserimento}?
    \begin{itemize}
        \item[\textbf{Sì:}] Gli eventi sono \textbf{indipendenti}. Per calcolare la probabilità di una sequenza, moltiplica le probabilità di ogni singola estrazione (che rimangono costanti).
    \end{itemize}
    \begin{itemize}
        \item[\textbf{No:}] Gli eventi sono \textbf{dipendenti}. Passa alla prossima domanda.
    \end{itemize}
    \vspace{5pt}
    \item[\textbf{Q2:}] L'**ordine di estrazione conta**?
    \begin{itemize}
        \item[\textbf{Sì}] (Parole chiave: "in sequenza", "prima... e poi...", "(nera, bianca)"):
        \begin{itemize}
            \item Usa la \textbf{Probabilità Condizionata} (Regola del Prodotto). Vedi card \textit{Estrazioni}.
        \end{itemize}
        \item[\textbf{No}] (Parole chiave: "in blocco", "un gruppo di", "2 nere e 1 bianca"):
        \begin{itemize}
            \item Usa le \textbf{Combinazioni} (Distribuzione Ipergeometrica). Vedi card \textit{Ipergeometrica}.
        \end{itemize}
    \end{itemize}
\end{itemize}
\end{card}

\section*{Sezione 1: Fondamenti di Probabilità}

\begin{card}[frametitle={Probabilità di Base}]
\textbf{Teoria:} Si usa per problemi semplici con un numero finito di esiti equiprobabili.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto: P(almeno un 6 in 3 lanci)} \\
    \hline
    \textbf{1. (Pro Tip) Usa il Complementare:} Se la domanda contiene "almeno uno...", è più facile calcolare la probabilità del suo contrario, cioè "nessuno". & L'evento complementare è "non ottenere nessun 6 in 3 lanci". \\
    \hline
    \textbf{2. Calcola la prob. del singolo evento (complementare):} & La probabilità di NON fare 6 in un singolo lancio è $5/6$. \\
    \hline
    \textbf{3. Calcola la prob. della sequenza:} Poiché i lanci sono indipendenti, moltiplichiamo le probabilità per ogni prova. & $P(\text{nessun 6 in 3 lanci}) = (\frac{5}{6})^3 = \frac{125}{216}$. \\
    \hline
    \textbf{4. Applica la formula del complementare:} Usa la formula $P(A) = 1 - P(A^c)$. & $P(\text{almeno un 6}) = 1 - \frac{125}{216} = \frac{91}{216}$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Probabilità Condizionata e Teorema di Bayes}]
\textbf{Teoria:} Permette di aggiornare la probabilità di una causa alla luce di un effetto osservato.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Test Medico)} \\
    \hline
    \textbf{1. Definisci Eventi e Dati:} Traduci il testo in notazione formale, identificando Cause ed Effetto. & 
    \begin{itemize}
        \item Cause: Malato (M), Sano (S).
        \item Effetto: Positivo (T).
        \item $P(M)=0.01 \implies P(S)=0.99$.
        \item $P(T|M)=0.95$, $P(T|S)=0.05$.
        \item \textbf{Obiettivo:} $P(M|T)$.
    \end{itemize} \\
    \hline
    \textbf{2. Calcola Denominatore (Prob. Totale):} Usa la Formula delle Probabilità Totali per trovare $P(\text{Effetto})$. \[ P(E) = \sum P(E|C_i)P(C_i) \] & 
    $P(T) = P(T|M)P(M) + P(T|S)P(S)$
    $= (0.95)(0.01) + (0.05)(0.99) = 0.059$. \\
    \hline
    \textbf{3. Calcola Numeratore:} Prendi i termini relativi solo alla causa che ti interessa: $P(E|C_k)P(C_k)$. & 
    $P(T|M)P(M) = (0.95)(0.01) = 0.0095$. \\
    \hline
    \textbf{4. Assembla la Formula di Bayes:} Dividi il Numeratore per il Denominatore. & 
    \[ P(M|T) = \frac{P(T|M)P(M)}{P(T)} = \frac{0.0095}{0.059} \approx 16.1\% \] \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\section*{Sezione 2: Variabili Aleatorie (V.A.)}

\begin{card}[frametitle={V.A. Generiche: Media e Varianza}]
\textbf{Teoria:} Calcola gli indici di posizione (media) e dispersione (varianza) per una v.a. generica.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto ($X$ con $P(0)=0.5, P(1)=0.3, P(2)=0.2$)} \\
    \hline
    \textbf{1. Elenca i valori e le probabilità:} Prepara una lista dei valori $k$ che la v.a. può assumere e delle loro probabilità $P(X=k)$. & Valori: $\{0, 1, 2\}$. \newline Probabilità: $\{0.5, 0.3, 0.2\}$. \\
    \hline
    \textbf{2. Calcola $E[X]$:} Calcola la somma pesata: \[ E[X] = \sum_k k \cdot P(X=k) \] & $E[X] = (0 \cdot 0.5) + (1 \cdot 0.3) + (2 \cdot 0.2) = 0.7$. \\
    \hline
    \textbf{3. Calcola $E[X^2]$:} Calcola la somma pesata dei quadrati: \[ E[X^2] = \sum_k k^2 \cdot P(X=k) \] & $E[X^2] = (0^2 \cdot 0.5) + (1^2 \cdot 0.3) + (2^2 \cdot 0.2) = 1.1$. \\
    \hline
    \textbf{4. Calcola la Varianza:} Usa la formula: \[ \text{Var}(X) = E[X^2] - (E[X])^2 \] & $\text{Var}(X) = 1.1 - (0.7)^2 = 1.1 - 0.49 = 0.61$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Probabilità di Eventi su V.A. (Casi Astratti)}]
\textbf{Teoria:} Calcolare la probabilità di un evento definito da una relazione tra v.a. (es. $P(X \ge Y)$).

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Calcolare $P(X_1=X_2)$)} \\
    \hline
    \textbf{1. Identifica i Casi Favorevoli:} Trova tutte le coppie/valori $(x,y,...)$ che soddisfano la condizione data. & L'evento "$X_1=X_2$" è l'insieme di tutte le coppie $(k,k)$ per $k=0, 1, 2, \dots$. \\
    \hline
    \textbf{2. Recupera le Probabilità:} Prendi i valori $p(x,y,...)$ per ogni caso favorevole dalla funzione di densità/probabilità fornita. & Per ogni coppia $(k,k)$, la probabilità è data dalla formula $p(k,k)$. \\
    \hline
    \textbf{3. Somma (o Integra):} Per v.a. discrete, somma tutte le probabilità trovate. \[ P(\text{Evento}) = \sum_{\text{casi favorevoli}} p(x,y) \] & La probabilità totale è la somma (in questo caso, una serie) di tutte le probabilità dei casi favorevoli: \[ P(X_1=X_2) = \sum_{k=0}^{\infty} p(k,k) \] \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\newpage

\begin{card}[frametitle={Valore Atteso e Varianza della Somma}]
\textbf{Teoria:} Fornisce le proprietà di media e varianza di una somma di due variabili aleatorie.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Somma dei punteggi di 2 dadi)} \\
    \hline
    \textbf{1. Calcola le Medie:} Trova $E[X]$ e $E[Y]$ e sommali. & $E[X]=E[Y]=3.5$. \newline $E[X+Y] = 3.5+3.5=7$. \\
    \hline
    \textbf{2. Calcola le Varianze:} Trova $\text{Var}(X)$, $\text{Var}(Y)$. & $\text{Var}(X)=\text{Var}(Y)=35/12$. \\
    \hline
    \textbf{3. Valuta la Covarianza:} Se le v.a. sono indipendenti, $\text{Cov}(X,Y)=0$. & I lanci dei dadi sono indipendenti, quindi $\text{Cov}(X,Y)=0$. \\
    \hline
    \textbf{4. Applica la Formula per la Varianza:} $Var(X+Y) = Var(X)+Var(Y)+2Cov(X,Y)$. & $\text{Var}(X+Y)=\frac{35}{12}+\frac{35}{12}+0=\frac{70}{12}$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Variabili Indicatrici ($I_A$)}]
\textbf{Teoria:} Metodo potente per calcolare il valore atteso di una v.a. che conta dei successi.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (N. medio di Assi in 5 carte)} \\
    \hline
    \textbf{1. Definisci gli Indicatori:} Scrivi $X$ come somma di $I_{A_i}$, dove $I_{A_i}=1$ se l'i-esimo evento è un successo. & Sia $I_i=1$ se la i-esima carta è un Asso. La v.a. $X$ che conta gli Assi è: \[ X = I_1+I_2+I_3+I_4+I_5 \] \\
    \hline
    \textbf{2. Calcola $P(A_i)$:} Trova la probabilità del singolo successo. & La probabilità che una specifica carta sia un Asso è $P(A_i) = 4/52$. \\
    \hline
    \textbf{3. Usa la Linearità:} Il valore atteso della somma è la somma dei valori attesi: \[ E[X] = \sum E[I_{A_i}] = \sum P(A_i) \] & \[ E[X] = \sum_{i=1}^5 P(A_i) = 5 \times \frac{4}{52} = \frac{20}{52} \] \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Variabili Aleatorie Congiunte}]
\textbf{Teoria:} Descrivono il comportamento simultaneo di due v.a. discrete tramite una tabella.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Data una tabella $p(x,y)$)} \\
    \hline
    \textbf{1. Calcolare le Marginali:} Per trovare $P(X=x)$, \textbf{somma lungo la riga}. Per $P(Y=y)$, \textbf{somma lungo la colonna}. & Per trovare $p_X(0)$, sommo tutti i valori sulla riga $x=0$. \\
    \hline
    \textbf{2. Verificare l'Indipendenza:} Trova \textbf{una sola cella} in cui $P(x,y) \neq P(x) \cdot P(y)$ per dimostrare la dipendenza. & Testo la cella (0,0): calcolo $p_X(0) \cdot p_Y(0)$ e lo confronto con il valore $p(0,0)$ dato nella tabella. Se sono diversi $\implies$ dipendenti. \\
    \hline
    \textbf{3. Calcolare P(Evento):} Somma le probabilità delle celle che soddisfano la condizione. & Per calcolare $P(X \ge Y)$, sommo le probabilità di tutte le celle $(x,y)$ in cui il valore di $x$ è maggiore o uguale a quello di $y$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Trasformazioni di V.A. ($Y=g(X)$)}]
\textbf{Teoria:} Serve a trovare la funzione di distribuzione di una funzione di una v.a.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto ($Y=X^2$ con $X \sim U(0,2)$)} \\
    \hline
    \textbf{1. Partire dalla CDF di Y:} Scrivi la sua definizione e sostituisci $Y$ con $g(X)$. \[ F_Y(y) = P(Y \le y) \] & $F_Y(y) = P(X^2 \le y)$. \\
    \hline
    \textbf{2. Isolare X:} Manipola la disequazione per ottenere $P(X \le h(y))$. Questo è uguale a $F_X(h(y))$. & Poiché $X \ge 0$, la disequazione diventa $P(X \le \sqrt{y})$. Quindi $F_Y(y) = F_X(\sqrt{y})$. \\
    \hline
    \textbf{3. Usare la CDF di X:} Trova la formula per la CDF della variabile di partenza, $F_X(x)$, e sostituisci. & Per una U(0,2), la CDF è $F_X(t)=t/2$. Sostituendo $t=\sqrt{y}$ otteniamo: \[ F_Y(y) = \frac{\sqrt{y}}{2} \] \\
    \hline
    \textbf{4. Derivare (per trovare la PDF):} Calcola $f_Y(y) = F'_Y(y)$. & \[ f_Y(y) = \frac{d}{dy}\left(\frac{\sqrt{y}}{2}\right) = \frac{1}{4\sqrt{y}} \] \\
    \hline
    \textbf{5. Trovare il nuovo dominio:} Applica la funzione $g(X)$ agli estremi del dominio di X. & Se $X \in [0,2]$, allora $Y=X^2$ appartiene a $[0^2, 2^2] = [0,4]$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\section*{Sezione 3: Modelli di Variabili Aleatorie}

\begin{card}[frametitle={Distribuzione Binomiale $\boldsymbol{X \sim \text{Bin}(n, p)}$}]
\textbf{Teoria:} Conta il numero di successi in $n$ prove binarie indipendenti.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Arciere)} \\
    \hline
    \textbf{1. Identifica i parametri:} Trova $n$ (numero prove), $p$ (prob. successo) e $k$ (successi richiesti). & Dati: $n=10$ tiri, $p=0.8$ di colpire, $k=7$ successi richiesti. \\
    \hline
    \textbf{2. Applica la formula:} \[ p_X(k) = \binom{n}{k} p^k (1-p)^{n-k} \] & \[ P(X=7) = \binom{10}{7} (0.8)^7 (0.2)^3 \approx 0.2013 \] \\
    \hline
    \textbf{3. Media e Varianza (se richieste):} Usa le formule $E[X]=np$ e $Var(X)=np(1-p)$. & $E[X]=10 \cdot 0.8 = 8$. \newline $Var(X)=10 \cdot 0.8 \cdot 0.2 = 1.6$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Distribuzione Ipergeometrica}]
\textbf{Teoria:} Si usa per estrazioni \textbf{senza reinserimento} da una popolazione divisa in gruppi.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Urna con 4B e 3N)} \\
    \hline
    \textbf{1. Identifica i parametri:} Trova $N$ (totale), $n$ (estratti), $K_i$ (dimensione gruppi), $k_i$ (estratti per gruppo). & Dati: $N=7$, $n=4$ estratti. Gruppi: $K_B=4$, $K_N=3$. Richiesta: $k_B=2$, $k_N=2$. \\
    \hline
    \textbf{2. Assembla la frazione:} La probabilità è data da Casi Favorevoli / Casi Possibili. \[ P(\text{evento}) = \frac{\binom{K_1}{k_1} \binom{K_2}{k_2} \cdots}{\binom{N}{n}} \] & \[ P(\text{2B, 2N}) = \frac{\binom{4}{2} \binom{3}{2}}{\binom{7}{4}} = \frac{6 \cdot 3}{35} = \frac{18}{35} \] \\
    \hline
\end{tabularx}
\end{center}
\end{card}


\begin{card}[frametitle={Estrazioni: Con Ordine vs Senza Ordine}]
\textbf{Teoria:} Distingue tra problemi in cui l'ordine di estrazione conta e quelli in cui non conta.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.8}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Modalità di Estrazione} & \textbf{Esempio Svolto (Urna: 4 Bianche, 3 Nere)} \\
    \hline
    \textbf{1. Senza Ordine ("in blocco"):} Si usano le combinazioni (Ipergeometrica). La formula è $\frac{\text{Casi Favorevoli}}{\text{Casi Possibili}}$. & \textbf{Domanda:} Estraggo 4 palline. $P(\text{2B e 2N})$? \newline \textbf{Soluzione:} \[ P = \frac{\binom{4}{2} \binom{3}{2}}{\binom{7}{4}} = \frac{6 \cdot 3}{35} = \frac{18}{35} \] \\
    \hline
    \textbf{2. Con Ordine ("una alla volta"):} Si usa la probabilità condizionata. La formula è $P(A \text{ e poi } B) = P(A) \cdot P(B|A)$. & \textbf{Domanda:} Estraggo 2 palline. $P(\text{1° N, 2° B})$? \newline \textbf{Soluzione:} \[ P = P(\text{1°N}) \cdot P(\text{2°B|1°N}) = \frac{3}{7} \cdot \frac{4}{6} = \frac{2}{7} \] \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Distribuzione Geometrica $\boldsymbol{X \sim \text{Geo}(p)}$}]
\textbf{Teoria:} Conta il numero di prove necessarie per ottenere il **primo successo**.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Lancio di un dado)} \\
    \hline
    \textbf{1. Identifica i parametri:} Trova $p$ (prob. successo) e $k$ (il tentativo del primo successo). & Domanda: Prob. che il primo 6 esca al 4° lancio? \newline Dati: $p=1/6$, $k=4$. \\
    \hline
    \textbf{2. Applica la formula:} \[ p_X(k) = (1-p)^{k-1}p \] & \[ P(X=4) = (1-1/6)^{4-1} \cdot (1/6) = (5/6)^3 \cdot (1/6) \approx 0.096 \] \\
    \hline
    \textbf{3. Media (se richiesta):} Usa la formula $E[X] = 1/p$. & Il numero medio di lanci per ottenere un 6 è $E[X] = \frac{1}{1/6} = 6$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Distribuzione di Poisson $\boldsymbol{X \sim Po(\lambda)}$}]
\textbf{Teoria:} Conta il numero di eventi rari in un intervallo fissato di tempo o spazio.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Call Center)} \\
    \hline
    \textbf{1. Identifica $\lambda$:} Trova il tasso medio di eventi nell'intervallo di tempo/spazio dato. & Dati del problema: Un call center riceve in media 3 chiamate/ora ($\lambda=3$). \\
    \hline
    \textbf{2. Adatta $\lambda$ (se necessario):} Se l'intervallo della domanda è diverso, riproporziona $\lambda$. & Domanda: Prob. di 0 chiamate in 20 minuti? \newline 20 min = 1/3 di ora. Il nuovo tasso è $\lambda' = 3 \cdot (1/3)=1$. \\
    \hline
    \textbf{3. Identifica $k$ e applica la formula:} Trova il numero di eventi richiesto e usa $P(X=k) = e^{-\lambda}\frac{\lambda^k}{k!}$. & La domanda chiede $k=0$ chiamate. \[ P(X=0) = e^{-1}\frac{1^0}{0!} = e^{-1} \approx 0.368 \] \\
    \hline
\end{tabularx}
\end{center}

\begin{card}[frametitle={Distribuzione Uniforme $\boldsymbol{X \sim U(a, b)}$}]
\textbf{Teoria:} Modella una scelta casuale in un intervallo continuo $[a,b]$.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto ($X \sim U(4,10)$)} \\
    \hline
    \textbf{1. Identifica gli estremi:} Trova $a$ e $b$ dell'intervallo. & Dati: $a=4, b=10$. \\
    \hline
    \textbf{2. Calcola l'Altezza:} L'altezza della densità (rettangolo) è $h = \frac{1}{b-a}$. & L'altezza della densità è $h = \frac{1}{10-4} = \frac{1}{6}$. \\
    \hline
    \textbf{3. Calcola la Base:} È la lunghezza dell'intervallo della domanda. & Domanda: $P(X>8)$. L'intervallo di interesse è $[8,10]$, quindi la Base è $10-8=2$. \\
    \hline
    \textbf{4. Calcola l'Area:} Probabilità = Base $\times$ Altezza. & Area = $2 \times \frac{1}{6} = \frac{1}{3}$. \\
    \hline
\end{tabularx}

    \textbf{2. Usa la formula di sopravvivenza:} Per calcolare $P(T > t)$, usa la scorciatoia: \[ P(T > t) = e^{-\lambda t} \] & Domanda: Prob. che duri più di 1200 ore ($t=1200$)? \[ P(T > 1200) = e^{-(1/1000) \cdot 1200} = e^{-1.2} \approx 0.301 \] \\

\begin{card}[frametitle={Distribuzione Normale $\boldsymbol{X \sim N(\mu, \sigma^2)}$}]

    \hline
    \textbf{1. Identifica i parametri:} Trova la media $\mu$ e la \textbf{deviazione standard $\sigma$} (ricorda $\sigma = \sqrt{\sigma^2}$). & Dati del problema: $\mu=175$, $\sigma=7$. \\
    \hline
    \textbf{2. Standardizza:} Trasforma il valore $x$ del problema in un valore $z$ usando la formula: \[ \boldsymbol{Z = \frac{X - \mu}{\sigma}} \] & \[ Z = \frac{185 - 175}{7} \approx 1.43 \] \\
    \hline
    \textbf{3. Riformula la probabilità e usa le tavole:} Riscrivi la domanda in termini di $Z$ e usa le regole di simmetria e la tavola $\Phi(z)$. & La domanda diventa $P(Z > 1.43)$. Usando la regola della coda destra: \[ P(Z > 1.43) = 1 - \Phi(1.43) = 1 - 0.9236 = 0.0764 \] \\

\section*{Sezione 4: Teoremi e Approssimazioni}

\begin{card}[frametitle={Teorema del Limite Centrale (TLC)}]
\textbf{Teoria:} La somma o media di un grande numero ($n>30$) di v.a. i.i.d. si distribuisce come una Normale.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto (Carico ascensore)} \\
    \hline
    \textbf{1. Analizza la singola v.a. $X_i$:} Calcola la sua media $\mu$ e varianza $\sigma^2$. & Dati per 1 persona: $\mu=75$ kg, $\sigma^2=100$. \\
    \hline
    \textbf{2. Calcola i parametri della Somma $S_n$:} Usa $E[S_n] = n\mu$ e $\text{Var}(S_n) = n\sigma^2$. & Per $n=50$ persone: $E[S_{50}] = 50 \cdot 75 = 3750$. \newline $\text{Var}(S_{50}) = 50 \cdot 100 = 5000$. \\
    \hline
    \textbf{3. Approssima e Standardizza:} Tratta $S_n$ come una $N(n\mu, n\sigma^2)$ e calcola lo Z-score per il valore richiesto. & Domanda: $P(S_{50} > 3850)$? \[ Z = \frac{3850 - 3750}{\sqrt{5000}} \approx 1.41 \] \\
    \hline
    \textbf{4. Risolvi con le tavole:} Calcola la probabilità finale. & $P(Z > 1.41) = 1 - \Phi(1.41) \approx 1 - 0.9207 = 0.0793$. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Legge dei Grandi Numeri (LGN)}]
\textbf{Teoria:} La media campionaria $\bar{X}_n$ converge alla media vera $\mu$ per $n$ che tende all'infinito.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta Concettuale} & \textbf{Esempio Svolto (Guadagno del Casinò)} \\
    \hline
    \textbf{1. Calcola la Media Vera ($\mu$):} Trova il valore atteso $E[X]$ della singola azione. & Il guadagno atteso per il casinò da una singola giocata alla roulette è $E[X] = +0.027\text{€}$. \\
    \hline
    \textbf{2. Applica il Principio:} Concludi che per $n$ molto grande, la media osservata $\bar{X}_n$ sarà quasi certamente uguale alla media vera $\mu$. & Per la LGN, dopo milioni di giocate ($n$ grande), il guadagno medio \textit{osservato} dal casinò sarà $\approx 0.027\text{€}$ per giocata, garantendo un profitto. \\
    \hline
\end{tabularx}
\end{center}
\end{card}

\begin{card}[frametitle={Approssimazione della Binomiale alla Normale}]
\textbf{Teoria:} Se $n$ è grande, una Binomiale può essere approssimata da una Normale.

\vspace{5pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|X| >{\columncolor{lightyellow}}X|}
    \hline
    \textbf{Passo della Ricetta} & \textbf{Esempio Svolto ($P(X>210)$ in 400 lanci)} \\
    \hline
    \textbf{1. Verifica le condizioni:} Controlla che $np > 5$ e $n(1-p) > 5$. & $n=400, p=0.5 \implies np=200 > 5$. Condizioni verificate. \\
    \hline
    \textbf{2. Calcola i parametri della Normale:} $\mu = np$ e $\sigma^2 = np(1-p)$. & $\mu=200$, $\sigma^2=100$. La Binomiale è circa $N(200, 100)$. \\
    \hline
    \textbf{3. Applica la correzione di continuità:} Trasforma la domanda discreta in continua. & $P(X > 210)$ diventa $P(X \ge 211)$. Applichiamo la correzione: $P(Y \ge 211-0.5) = P(Y \ge 210.5)$. \\
    \hline
    \textbf{4. Standardizza e Risolvi:} Procedi come un normale problema sulla Gaussiana. & $Z = \frac{210.5-200}{\sqrt{100}} = 1.05$. \[ P(Z \ge 1.05) = 1 - \Phi(1.05) \] \\
    \hline
\end{tabularx}
\end{center}
\end{card}
